function prevM(){
	var prevM = month--;
	calendar(year,month);
	if(month<1){
		var lastM = 12;

		month = lastM;
		year = prevY;

		calendar(year, month);
	}
}



function nextM(){
	var nextM = month++;
	calendar(year,month);

		year = nextY;
